A.Implement simple linear regression using a dataset :
import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
from sklearn.linear_model import LinearRegression
from sklearn.metrics import mean_squared_error, r2_score
import seaborn as sns
import warnings
from sklearn.model_selection import train_test_split

#
df = pd.read_csv("marketing.csv") //change csv file according to paper

#
df.head()

#
df.shape

#
df.isnull().sum

#
df.dtypes

#
df.info

#
plt.figure(figsize=(10,5))
sns.pairplot(data=df, y_vars=['sales'], x_vars = ['youtube', 'facebook', 'newspaper' ])

#
 x= df[['newspaper']] #Predictor
y = df[['sales' ]] #Response
x_train, x_test, y_train, y_test = train_test_split(x,y, test_size = 0.2, random_state = 42)
model = LinearRegression()
model.fit(x_train, y_train)

B. Explore and interpret the regression model coefficients and goodness-offit measures :

y_test_pred = model.predict(x_test)
mse = mean_squared_error(y_test , y_test_pred)
print(mse)
r2_test = r2_score(y_test , y_test_pred)
print(r2_test)
print("Intercept:" , model.intercept_)
print("Slope (Coefficient):" , model.coef_[0])
