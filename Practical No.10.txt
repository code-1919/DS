colors = ["Red", "Blue"]

df_class = df["Attrition_numeric"].value_counts().reset_index()
df_class.columns = ["Attrition", "Frequency"]

plt.figure(figsize=(6,3))
ax = sns.barplot(
    x="Attrition",
    y="Frequency",
    data=df_class,
    palette=colors
)

plt.title("Attrition Frequency")
plt.xlabel("Attrition")
plt.ylabel("Frequency")

# show values on bars
for i, v in enumerate(df_class["Frequency"]):
    ax.text(i, v, v, ha="center", va="bottom")

sns.despine()
plt.show()
